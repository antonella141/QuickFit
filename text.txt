<script>
  const closetDisplay = document.getElementById('closetDisplay');
  const favoritesList = document.getElementById('favoritesList');
  const suggestionDisplay = document.getElementById('suggestionDisplay');
  const friendsList = document.getElementById('friendsList');

  const closetItems = [];
  const favorites = [];
  const friends = [];

  document.getElementById('addClothes').addEventListener('change', function(e){
    Array.from(e.target.files).forEach(file => {
      const reader = new FileReader();
      reader.onload = function(ev){
        const imgSrc = ev.target.result;
        closetItems.push(imgSrc);
        const card = document.createElement('div');
        card.className = 'card';
        card.innerHTML = `<img src="${imgSrc}"><br><button onclick="addToFavorites('${imgSrc}')">❤️ Favorite</button>`;
        closetDisplay.appendChild(card);
      }
      reader.readAsDataURL(file);
    });
  });

  function addToFavorites(src){
    if(!favorites.includes(src)){
      favorites.push(src);
      const card = document.createElement('div');
      card.className='card';
      card.innerHTML=`<img src="${src}">`;
      favoritesList.appendChild(card);
    }
  }

  function getSuggestion(){
    if(closetItems.length===0){
      suggestionDisplay.innerText = "Upload clothes first!";
      return;
    }
    const randomItem = closetItems[Math.floor(Math.random()*closetItems.length)];
    suggestionDisplay.innerHTML = `<div class="card"><img src="${randomItem}"><br>Suggested Outfit</div>`;
  }

  function addFriend(){
    const name = document.getElementById('friendName').value.trim();
    if(name && !friends.includes(name)){
      friends.push(name);
      const card = document.createElement('div');
      card.className='card';
      card.innerHTML=`${name}<br><button onclick="alert('Friend Request Approved!')">Approve</button>`;
      friendsList.appendChild(card);
    }
    document.getElementById('friendName').value='';
  }
</script>
